
RHS generation old FEAT:
------------------------

DO IEL = 1, NEL
  Get local/global DOF's on element IEL
  DO ICUB = 1,NCUB
    Compute values of the basis functions on the element
    DO IDOF = 1,NLDOF
      Calculate transformation: reference -> real element
      Calculate value of FE function in IDOF
      Get coefficient of function f in (f,phi)
      Assemble additive entry and put into RHS
    END DO
  END DO
END DO







RHS generation new FEAT:
------------------------

DO IELBLOCK = 1, NEL, 1000
  Calculate transformation for all cub.points in the 1000 elements
  Get all coefficients for all cub.points on the 1000 elements
  Get all local/global DOF's on these 1000 elements
  DO IEL = 1,1000
    DO ICUB = 1,NCUB
      DO IDOF = 1,NLDOF
        Assemble 'local' additive contributions for RHS vector
      END DO
    END DO
    Incorporate 'local' contributions to global vector
  END DO
END DO
